function svdfit_func, X, M

	 ;return, [ [1.0], [sin(X)], [cos(X)], [sin((2.0/1)*X)], [cos((2.0/1)*X)] ]
         return, [ [1.0], [sin((2.0/1)*!const.pi*X)], [cos((2.0/1)*!const.pi*X)], $
		 [sin((4.0/1)*!const.pi*X)], [cos((4.0/1)*!const.pi*X)] ]
end
